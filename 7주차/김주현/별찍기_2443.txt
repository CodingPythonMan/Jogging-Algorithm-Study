import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		
		Scanner input = new Scanner(System.in);
		int N = input.nextInt();
		input.close();
		
		for(int i =1 ; i <= N ;i++) {
			
			for(int j = 0 ; j < i; j++){	
				System.out.print(" ");
			}
			
			for(int k=i; k <= N*2-i  ;k++ ) {
				System.out.print("*");
			}
			
			System.out.println("");
		}
	
	}

}
