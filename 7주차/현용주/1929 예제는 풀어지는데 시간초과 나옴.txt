import java.util.*;

public class Main {

	public static void main(String[] args) {
		Scanner sc = new Scanner(System.in);

		int N = sc.nextInt();
		int M = sc.nextInt();
		
		for(int i = N; i <= M; i++) {
			for(int j = 2; j <= M; j++) {
				if(i != j && i % j == 0) {
					break;
				}
				if(i % j == 0) {
					System.out.println(i);
				}
			}
		}
		
	}

}
